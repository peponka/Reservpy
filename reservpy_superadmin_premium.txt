BUG #1 — FIX CRÍTICO: LOOP DE "ENTRAR COMO ADMINISTRADOR"
================================================================

Problema:
Después de hacer login, la pantalla sigue mostrando el botón
"Entrar como Administrador" en lugar de redirigir al dashboard.
El estado de autenticación no se propaga correctamente.

Causa probable:
El componente de login no espera a que el contexto de auth
(AuthContext o Zustand store) se actualice antes de decidir
el redirect. El guard de ruta chequea el rol antes de que
Supabase termine de devolver la sesión.

Solución completa:

1. En el AuthProvider, asegurarse de tener un estado "loading":

   const [user, setUser] = useState(null)
   const [role, setRole] = useState(null)
   const [loading, setLoading] = useState(true)

   useEffect(() => {
     supabase.auth.getSession().then(({ data: { session } }) => {
       setUser(session?.user ?? null)
       if (session?.user) {
         fetchUserRole(session.user.id).then(setRole)
       }
       setLoading(false)
     })

     const { data: { subscription } } = supabase.auth.onAuthStateChange(
       async (_event, session) => {
         setUser(session?.user ?? null)
         if (session?.user) {
           const r = await fetchUserRole(session.user.id)
           setRole(r)
         } else {
           setRole(null)
         }
       }
     )
     return () => subscription.unsubscribe()
   }, [])

2. En el guard de ruta del admin (middleware.ts o layout.tsx):
   - Si loading === true: mostrar skeleton, NO redirigir
   - Si loading === false y role === 'admin' o 'super_admin': permitir
   - Si loading === false y no hay sesión: redirigir a /login
   - Si ya hay sesión activa en /login: redirigir a /admin/dashboard

3. Eliminar cualquier botón manual de "Entrar como Administrador".
   El acceso debe ser automático basado en el rol, nunca manual.

-- TABLA: team_members
-- Miembros del equipo con roles granulares

  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id         UUID REFERENCES auth.users(id) ON DELETE SET NULL,
  email           TEXT NOT NULL UNIQUE,
  full_name       TEXT,
  role            TEXT NOT NULL
                    CHECK (role IN ('super_admin','admin','manager','soporte')),
  status          TEXT NOT NULL DEFAULT 'pending'
                    CHECK (status IN ('active','pending','revoked')),
  invite_token    TEXT UNIQUE,
  invite_expires  TIMESTAMPTZ,
  invited_by      UUID REFERENCES auth.users(id),
  invited_at      TIMESTAMPTZ DEFAULT now(),
  accepted_at     TIMESTAMPTZ,
  last_login_at   TIMESTAMPTZ,
  created_at      TIMESTAMPTZ DEFAULT now(),
  updated_at      TIMESTAMPTZ DEFAULT now()
);

-- VISTA: clientes con pagos vencidos (dashboard y centro de pagos)

CREATE VIEW v_overdue_clients AS
SELECT
  c.id                  AS client_id,
  c.business_name,
  c.email,
  c.phone,
  c.status              AS client_status,
  pl.name               AS plan_name,
  p.id                  AS payment_id,
  p.amount,
  p.currency,
  p.due_date,
  CURRENT_DATE - p.due_date AS days_overdue,
  CASE
    WHEN CURRENT_DATE - p.due_date BETWEEN 1 AND 7   THEN 'leve'
    WHEN CURRENT_DATE - p.due_date BETWEEN 8 AND 30  THEN 'moderado'
    WHEN CURRENT_DATE - p.due_date > 30              THEN 'critico'
  END AS severity,
  (
    SELECT MAX(sent_at) FROM payment_reminders pr
    WHERE pr.client_id = c.id AND pr.payment_id = p.id
  ) AS last_reminder_sent
FROM clients c
JOIN payments p  ON p.client_id = c.id
LEFT JOIN plans pl ON pl.id = c.plan_id
WHERE p.status = 'overdue'
  AND c.status NOT IN ('suspended','cancelled')
ORDER BY days_overdue DESC;


-- VISTA: métricas de ingresos mensuales (para gráfico del dashboard)

CREATE VIEW v_monthly_revenue AS
SELECT
  DATE_TRUNC('month', paid_at) AS month,
  COUNT(*)                      AS payment_count,
  SUM(amount)                   AS total_revenue,
  currency
FROM payments
WHERE status = 'paid'
  AND paid_at IS NOT NULL
GROUP BY DATE_TRUNC('month', paid_at), currency
ORDER BY month DESC;

-- Helper function: obtiene el rol del usuario autenticado
CREATE OR REPLACE FUNCTION get_my_role()
RETURNS TEXT AS $$
  SELECT role FROM team_members
  WHERE user_id = auth.uid() AND status = 'active'
  LIMIT 1;
$$ LANGUAGE sql SECURITY DEFINER;

-- Lectura de clientes: todo el equipo activo puede ver
CREATE POLICY "team_read_clients" ON clients
  FOR SELECT USING (
    get_my_role() IN ('super_admin','admin','manager','soporte')
  );

-- Escritura en clientes: solo admin y super_admin
CREATE POLICY "admin_write_clients" ON clients
  FOR ALL USING (
    get_my_role() IN ('super_admin','admin')
  );

-- Pagos: lectura para todo el equipo excepto soporte
CREATE POLICY "team_read_payments" ON payments
  FOR SELECT USING (
    get_my_role() IN ('super_admin','admin','manager')
  );

-- Pagos: escritura solo admin+
CREATE POLICY "admin_write_payments" ON payments
  FOR ALL USING (
    get_my_role() IN ('super_admin','admin')
  );

-- Equipo: solo super_admin y admin gestionan miembros
CREATE POLICY "admin_manage_team" ON team_members
  FOR ALL USING (
    get_my_role() IN ('super_admin','admin')
  );

-- Log: solo admins ven el log completo
CREATE POLICY "admin_read_log" ON team_activity_log
  FOR SELECT USING (
    get_my_role() IN ('super_admin','admin')
  );

-- Log: cualquier miembro autenticado puede insertar (auditoría)
CREATE POLICY "team_insert_log" ON team_activity_log
  FOR INSERT WITH CHECK (
    get_my_role() IS NOT NULL
  );
